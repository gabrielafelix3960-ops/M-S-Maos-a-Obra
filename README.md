    // Formulário: validação simples e suporte a Netlify Forms / Formspree / fallback mailto
    (function(){
      const form = document.getElementById('contactForm');
      const status = document.getElementById('formStatus');

      function showStatus(msg, color){
        status.textContent = msg;
        status.style.color = color || 'var(--dark-blue)';
      }

      form && form.addEventListener('submit', function(e){
        e.preventDefault();
        status.textContent = '';
        const name = document.getElementById('name').value.trim();
        const phone = document.getElementById('phone').value.trim();
        const email = document.getElementById('email').value.trim();
        const message = document.getElementById('message').value.trim();

        if(!name || !email || !message){
          showStatus('Por favor preencha os campos obrigatórios.', 'var(--orange)');
          return;
        }

        // Se houver meta tag com endpoint do Formspree, usar fetch para submeter
        const meta = document.querySelector('meta[name="formspree-endpoint"]');
        if(meta && meta.content){
          const endpoint = meta.content.trim();
          const payload = {
            name: name,
            phone: phone,
            email: email,
            message: message
          };
          showStatus('A enviar...');
          fetch(endpoint, {
            method: 'POST',
            headers: {'Content-Type': 'application/json', 'Accept': 'application/json'},
            body: JSON.stringify(payload)
          }).then(r => r.json()).then(data => {
            showStatus('Mensagem enviada. Obrigado!');
            form.reset();
          }).catch(err => {
            showStatus('Erro ao enviar. Tente enviar por email: msmaosaobraconstrucoes@gmail.com', 'var(--orange)');
            console.error(err);
          });
          return;
        }

        // Caso contrário, fallback: se o site for hospedado no Netlify, o atributo data-netlify="true" permite submissão automática sem JS
        // Se não houver Formspree e não estivermos a usar Netlify, abrimos o cliente de email (mailto) como fallback.
        const subject = encodeURIComponent('Contacto via site – ' + name);
        const bodyLines = [
          'Nome: ' + name,
          'Telefone: ' + phone,
          'Email: ' + email,
          '',
          'Mensagem:',
          message
        ];
        const body = encodeURIComponent(bodyLines.join('\n'));
        const mailto = 'mailto:msmaosaobraconstrucoes@gmail.com?subject=' + subject + '&body=' + body;
        showStatus('A abrir o cliente de email...');
        window.location.href = mailto;
      });
    })();
  </script>
